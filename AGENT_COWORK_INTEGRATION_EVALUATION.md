import { CacheManager } from './cache-manager';

interface RoutingResult {
  skill: Skill;
  confidence: number;
  reasoning: string;
}

export class SkillRouter {
  private index: SkillIndex;
  private cache: CacheManager;

  constructor() {
    this.index = new SkillIndex();
    this.cache = new CacheManager();
  }

  // æ„å›¾è·¯ç”±ï¼ˆæ ¸å¿ƒç®—æ³•ï¼‰
  async route(query: string): Promise<RoutingResult[]> {
    // æ£€æŸ¥ç¼“å­˜
    const cached = this.cache.get(`route:${query}`);
    if (cached) return cached;

    const results: RoutingResult[] = [];

    // Priority 1: ç²¾ç¡®å‘½ä»¤åŒ¹é…
    if (query.startsWith('/')) {
      const skill = this.index.findByCommand(query);
      if (skill) {
        results.push({ skill, confidence: 1.0, reasoning: 'Exact command match' });
        return results;
      }
    }

    // Priority 2: å…³é”®è¯åŒ¹é…ï¼ˆä½¿ç”¨åå‘ç´¢å¼•ï¼‰
    const keywordMatches = this.index.searchByKeywords(query);
    results.push(...keywordMatches.map(match => ({
      skill: match.skill,
      confidence: match.score,
      reasoning: `Keyword match: ${match.matchedKeywords.join(', ')}`
    })));

    // Priority 3: åˆ†ç±»åŒ¹é…
    const categoryMatches = this.index.searchByCategory(query);
    results.push(...categoryMatches);

    // Priority 4: åä½œé“¾æ¨ç†
    if (results.length > 0) {
      const collaborationChain = await this.findCollaborationChain(results[0].skill);
      results.push(...collaborationChain);
    }

    // ç¼“å­˜ç»“æœ
    this.cache.set(`route:${query}`, results);

    return results.sort((a, b) => b.confidence - a.confidence);
  }

  // åä½œé“¾æ¨ç†
  private async findCollaborationChain(skill: Skill): Promise<RoutingResult[]> {
    const chain: RoutingResult[] = [];
    const visited = new Set<string>();

    const traverse = (currentSkill: Skill, depth: number) => {
      if (depth > 3 || visited.has(currentSkill.id)) return;
      visited.add(currentSkill.id);

      // åŸºäº provides/consumes æ‰¾å…³è”æŠ€èƒ½
      const relatedSkills = this.index.findRelatedSkills(currentSkill);
      relatedSkills.forEach(related => {
        chain.push({
          skill: related,
          confidence: 0.7 - (depth * 0.1),
          reasoning: `Collaboration chain from ${currentSkill.name}`
        });
        traverse(related, depth + 1);
      });
    };

    traverse(skill, 0);
    return chain;
  }
